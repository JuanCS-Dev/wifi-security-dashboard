def render(self) -> str:
    """
    â•”â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•—
    â•‘             GRID RENDERING ALGORITHM                     â•‘
    â•šâ•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•

    Time Complexity: O(n Ã— m)
      where n = number of components
            m = average lines per component

    Space Complexity: O(w Ã— h)
      where w = terminal width
            h = terminal height
    """
    output = []

    # Step 1: Clear screen and reset cursor
    output.append("\033[2J\033[H")           # O(1)

    # Step 2: For each component
    for component in self._components:       # O(n)

        # Step 3: Render component to string buffer
        lines = self._render_component(      # O(m)
            component.renderable,
            component.width,
            component.height
        )

        # Step 4: Position each line with ANSI codes
        for line_offset, line in enumerate(lines):  # O(m)
            row = component.y + line_offset + 1
            col = component.x + 1

            # Skip if out of bounds
            if row > self.height or row < 1:
                continue

            # Append ANSI-positioned line
            output.append(f"\033[{row};{col}H{line}")

    # Step 5: Move cursor to bottom (cleanup)
    output.append(f"\033[{self.height};1H")  # O(1)

    return ''.join(output)                   # O(total_chars)
